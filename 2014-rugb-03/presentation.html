# Gemfile
source 'https://rubygems.org'
gem 'slippery'

# Rakefile
require 'slippery'
Slippery::RakeTasks.new

echo "# Let's see" > talk.md
rake slippery:build:talk

Slippery::RakeTasks.new do
  self.options = {
    type: :reveal_js,
    theme: 'sky',
    transition: 'slide',
  }

  title 'Slippery Slides'
  add_highlighting
  self_contained
end
